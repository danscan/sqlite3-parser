Example Queries (Click to load):

Simple SELECT
SELECT * FROM users
Complex JOIN with Aggregation
SELECT u.id, u.name, COUNT(p.id) as post_count FROM users u LEFT JOIN posts p ON u.id = p.user_id WHERE u.active = true GROUP BY u.id, u.name ORDER BY post_count DESC
INSERT with Parameters
INSERT INTO users (name, email, active) VALUES (?, ?, true)
UPDATE with SQLite Functions
UPDATE users SET active = false WHERE last_login < datetime("now", "-1 year")
CREATE TABLE
CREATE TABLE posts (id INTEGER PRIMARY KEY, title TEXT NOT NULL, content TEXT, user_id INTEGER REFERENCES users(id))
CTE (Common Table Expression)
WITH recent_posts AS (SELECT * FROM posts WHERE created_at > datetime("now", "-1 month")) SELECT u.name, COUNT(rp.id) FROM users u JOIN recent_posts rp ON u.id = rp.user_id GROUP BY u.id
Mixed Read/Write Access
UPDATE users SET last_active = datetime("now") WHERE id IN (SELECT user_id FROM sessions WHERE created_at > datetime("now", "-1 hour"))
